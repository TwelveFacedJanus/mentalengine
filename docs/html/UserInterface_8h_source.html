229 ImGui::Begin("Console");
230
231 // Кнопки управления
232 if (ImGui::Button("Clear")) {
233 std::lock_guard<std::mutex> lock(console_mutex);
234 console_output.clear();
235 }
236 ImGui::SameLine();
237 if (ImGui::Button("Scroll to Bottom")) {

240
241 ImGui::Separator();
242
243 // Область вывода терминала
244 ImGui::BeginChild("ConsoleOutput", ImVec2(0, -ImGui::GetFrameHeight() - 10), true);
245
246 // Отображаем вывод терминала
247 {
248 std::lock_guard<std::mutex> lock(console_mutex);

250 ImGui::TextUnformatted(line.c_str());

252 }
253
254 // Автоматическая прокрутка вниз
255 if (console_scroll_to_bottom) {
256 ImGui::SetScrollHereY(1.0f);

258 }
259
260 ImGui::EndChild();
261
262 // Поле ввода команды
263 ImGui::PushItemWidth(-1);
264 static char input_buf[256] = "";
265 if (ImGui::InputText("##ConsoleInput", input_buf, sizeof(input_buf), ImGuiInputTextFlags_EnterReturnsTrue)) {
266 if (strlen(input_buf) > 0) {
267 std::string command = std::string(input_buf);
268 // Добавляем команду в вывод
269 __add_console_output("> " + command);
270
271 // Здесь можно добавить логику выполнения команды
272 if (command == "clear") {
273 std::lock_guard<std::mutex> lock(console_mutex);
274 console_output.clear();
275 } else if (command == "help") {
276 __add_console_output("Доступные команды:");
277 __add_console_output(" clear - очистить консоль");
278 __add_console_output(" help - показать эту справку");
279 __add_console_output(" quit - выйти из приложения");
280 } else if (command == "quit") {
281 __add_console_output("Выход из приложения...");
282 // Здесь можно добавить логику выхода
283 } else {
284 __add_console_output("Неизвестная команда: " + command);
285 }
286
287 input_buf[0] = '\0';
288 console_scroll_to_bottom = true;
289 }
290 }
291 ImGui::PopItemWidth();
292
293 // Фокус на поле ввода
294 if (ImGui::IsWindowAppearing()) {
295 ImGui::SetKeyboardFocusHere(-1);
296 }
297
298 ImGui::End();

618template <typename T>
619nil UserInterface<T>::__load_default_fonts() {
620 // Загружаем SF Pro Text как основной шрифт
621 const char* fontPath = "resources/fonts/SFProText-Regular.ttf";
622
623 // Проверяем существование файла
624 FILE* file = fopen(fontPath, "r");
625 if (!file) {
626 std::cerr << "Предупреждение: Не удалось найти шрифт " << fontPath << ", используется шрифт по умолчанию" << std::endl;
627 return;
628 }
629 fclose(file);
630
631 // Очищаем существующие шрифты
632 this->pIO->Fonts->Clear();
633
634 // Загружаем SF Pro Text с разными размерами
635 ImFont* sfProFont16 = this->pIO->Fonts->AddFontFromFileTTF(
636 fontPath,
637 16.0f, // Размер шрифта в пикселях
638 nullptr, // Конфигурация шрифта (опционально)
639 this->pIO->Fonts->GetGlyphRangesDefault() // Диапазоны символов
640 );
641
642 // Загружаем дополнительный размер для заголовков
643 ImFont* sfProFont24 = this->pIO->Fonts->AddFontFromFileTTF(
644 fontPath,
645 24.0f, // Больший размер для заголовков
646 nullptr,
647 this->pIO->Fonts->GetGlyphRangesDefault()
648 );
649
650 if (sfProFont16) {
651 this->pIO->FontDefault = sfProFont16;
652 std::cout << "Шрифт SF Pro Text (16px) успешно загружен" << std::endl;
653 if (sfProFont24) {
654 std::cout << "Шрифт SF Pro Text (24px) также загружен" << std::endl;
655 }
656 } else {
657 std::cerr << "Ошибка загрузки шрифта SF Pro Text" << std::endl;
658 // Загружаем шрифт по умолчанию как fallback
659 this->pIO->Fonts->AddFontDefault();
660 }
661
662 // Собираем атлас шрифтов
663 this->pIO->Fonts->Build();
664}

718template <typename T>
719nil UserInterface<T>::__add_console_output(const std::string& text) {
720 std::lock_guard<std::mutex> lock(console_mutex);
721
722 // Разбиваем текст на строки
723 std::istringstream iss(text);
724 std::string line;
725 while (std::getline(iss, line)) {
726 console_output.push_back(line);
727
728 // Ограничиваем количество строк
729 if (console_output.size() > MAX_CONSOLE_LINES) {
730 console_output.erase(console_output.begin());
731 }
732 }
733
734 console_scroll_to_bottom = true;
735}
